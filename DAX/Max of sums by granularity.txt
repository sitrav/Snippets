# returns highest granularity summed value# [Qty] is a measure sum or count# [Month] is the granularity example# filters out blank values=calculate(	maxx( 		values( tbl[Month] ),		[Qty] 	),	filter( 		tbl,		[Qty] 	))#returns the Granularity entry for the Max Value (aka month name: Jan, Feb,...)=IF( 	ISBLANK( [Max of sums by granularity] ),	BLANK(),	CALCULATE( 		FIRSTNONBLANK( 			tbl[Month],			tbl[Month]		),		FILTER( 			VALUES( tbl[Month] ),			[Qty] = 			CALCULATE(				[Max of sums by granularity],				VALUES( tbl[Month] )			)		)	))
